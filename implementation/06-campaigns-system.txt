3. useCampaigns() HOOK
============================================================

Location: src/hooks/useCampaigns.ts

  function useCampaigns() {
    const { user } = useAuth();
    const [campaigns, setCampaigns] = useState([]);
    const [loading, setLoading] = useState(true);

    // Fetch all campaigns
    const fetchCampaigns = async () => {
      const { data, error } = await supabase
        .from("campaigns")
        .select("*")
        .eq("user_id", user.id)
        .order("created_at", { ascending: false });
      if (!error) setCampaigns(data);
      setLoading(false);
    };

    // Create campaign
    const createCampaign = async (campaign) => {
      const { data, error } = await supabase
        .from("campaigns")
        .insert({ ...campaign, user_id: user.id })
        .select()
        .single();
      if (!error) setCampaigns(prev => [data, ...prev]);
      return { data, error };
    };

    // Update campaign
    const updateCampaign = async (id, updates) => {
      const { data, error } = await supabase
        .from("campaigns")
        .update(updates)
        .eq("id", id)
        .select()
        .single();
      if (!error) {
        setCampaigns(prev => prev.map(c => c.id === id ? data : c));
      }
      return { data, error };
    };

    // Delete campaign
    const deleteCampaign = async (id) => {
      const { error } = await supabase
        .from("campaigns")
        .delete()
        .eq("id", id);
      if (!error) {
        setCampaigns(prev => prev.filter(c => c.id !== id));
      }
      return { error };
    };

    // Toggle pause/resume
    const toggleStatus = async (id, currentStatus) => {
      const newStatus = currentStatus === "active" ? "paused" : "active";
      return updateCampaign(id, { status: newStatus });
    };

    // Duplicate campaign
    const duplicateCampaign = async (campaign) => {
      const { id, created_at, updated_at, budget_spent, ...rest } = campaign;
      return createCampaign({
        ...rest,
        name: campaign.name + " (Copy)",
        status: "draft",
        budget_spent: 0,
      });
    };

    useEffect(() => {
      if (user) fetchCampaigns();
    }, [user]);

    return {
      campaigns,
      loading,
      createCampaign,
      updateCampaign,
      deleteCampaign,
      toggleStatus,
      duplicateCampaign,
      refetch: fetchCampaigns,
    };
  }

4.1 ChannelBadge.tsx
---------------------

  Props: channel: "meta" | "google" | "x" | "email"

  Renders: icon + label in a pill

  Icons:
    - meta: Facebook icon (custom SVG or 📱)
    - google: Search icon (🔍 or custom)
    - x: X/Twitter icon (custom SVG)
    - email: Mail icon from lucide

  Colors:
    - meta: bg-blue-50 text-blue-700
    - google: bg-amber-50 text-amber-700
    - x: bg-slate-100 text-slate-700
    - email: bg-purple-50 text-purple-700


4.2 StatusBadge.tsx
--------------------

  Props: status: "active" | "paused" | "draft" | "completed"

  Renders: dot + label in a pill

  Colors:
    - active: green dot, bg-emerald-50 text-emerald-700
    - paused: amber dot, bg-amber-50 text-amber-700
    - draft: gray dot, bg-slate-100 text-slate-500
    - completed: blue dot, bg-blue-50 text-blue-700
