DATABASE SCHEMA (SUPABASE)
===========================

This file contains the complete SQL schema for all dashboard tables,
RLS policies, triggers, indexes, and seed data.

Run this in the Supabase SQL Editor (Dashboard → SQL Editor → New Query).


============================================================
PART 1: NEW TABLES
============================================================

-- -------------------------------------------------------
-- 1. CAMPAIGNS
-- The core table. Every marketing campaign lives here.
-- -------------------------------------------------------
CREATE TABLE public.campaigns (
  id              UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id         UUID REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL,

  -- Basic info
  name            TEXT NOT NULL,
  objective       TEXT NOT NULL,               -- 'awareness' | 'leads' | 'conversions' | 'traffic'
  channel         TEXT NOT NULL,               -- 'meta' | 'google' | 'x' | 'email'
  status          TEXT NOT NULL DEFAULT 'draft', -- 'draft' | 'active' | 'paused' | 'completed'

  -- Audience targeting
  audience_persona_id UUID REFERENCES public.audience_personas(id) ON DELETE SET NULL,
  audience_custom TEXT,                        -- free-text audience description if not using a persona
  target_age_min  INT DEFAULT 18,
  target_age_max  INT DEFAULT 65,
  target_location TEXT,
  target_interests TEXT[],                     -- ['saas', 'ai', 'startups']

  -- AI copy generation
  tone            TEXT,                        -- 'bold' | 'witty' | 'professional' | 'inspiring'
  product_name    TEXT,
  product_description TEXT,
  generated_headline TEXT,                     -- AI-generated headline
  generated_body  TEXT,                        -- AI-generated ad body copy
  generated_cta   TEXT,                        -- AI-generated CTA text
  generated_image_url TEXT,                    -- URL to Gemini-generated ad image (stored in Supabase Storage)

  -- Budget & schedule
  daily_budget    DECIMAL(10,2),
  total_budget    DECIMAL(10,2),
  budget_spent    DECIMAL(10,2) DEFAULT 0,
  bid_strategy    TEXT DEFAULT 'lowest_cost',  -- 'lowest_cost' | 'target_cpa' | 'target_roas'
  start_date      DATE,
  end_date        DATE,                        -- NULL = run continuously

  -- AI predictions
  ai_confidence_score INT,                     -- 0-100
  predicted_ctr   DECIMAL(5,2),                -- e.g. 3.20 = 3.2%
  predicted_cpa   DECIMAL(10,2),               -- e.g. 4.50

  -- Timestamps
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);

-- Indexes for common queries
CREATE INDEX idx_campaigns_user_id ON public.campaigns(user_id);
CREATE INDEX idx_campaigns_status ON public.campaigns(status);
CREATE INDEX idx_campaigns_channel ON public.campaigns(channel);
CREATE INDEX idx_campaigns_created_at ON public.campaigns(created_at DESC);


-- -------------------------------------------------------
-- 2. CAMPAIGN METRICS (daily snapshots)
-- One row per campaign per day. Powers all charts.
-- -------------------------------------------------------
CREATE TABLE public.campaign_metrics (
  id              UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  campaign_id     UUID REFERENCES public.campaigns(id) ON DELETE CASCADE NOT NULL,
  date            DATE NOT NULL,

  -- Core metrics
  impressions     INT DEFAULT 0,
  clicks          INT DEFAULT 0,
  conversions     INT DEFAULT 0,
  spend           DECIMAL(10,2) DEFAULT 0,     -- actual spend that day
  revenue         DECIMAL(10,2) DEFAULT 0,     -- attributed revenue that day

  -- Calculated metrics (stored for fast reads)
  ctr             DECIMAL(5,4) DEFAULT 0,      -- clicks / impressions
  cpa             DECIMAL(10,2) DEFAULT 0,     -- spend / conversions
  roas            DECIMAL(5,2) DEFAULT 0,      -- revenue / spend

  created_at      TIMESTAMPTZ DEFAULT NOW(),

  -- Prevent duplicate entries for same campaign+date
  UNIQUE(campaign_id, date)
);

CREATE INDEX idx_metrics_campaign_date ON public.campaign_metrics(campaign_id, date DESC);


-- -------------------------------------------------------
-- 3. AUDIENCE PERSONAS
-- Saved audience profiles, created manually or via Gemini AI.
-- -------------------------------------------------------
CREATE TABLE public.audience_personas (
  id              UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id         UUID REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL,

  name            TEXT NOT NULL,                -- e.g. "Tech Founders"
  description     TEXT,                         -- detailed description
  age_min         INT DEFAULT 18,
  age_max         INT DEFAULT 65,
  gender          TEXT DEFAULT 'all',           -- 'all' | 'male' | 'female' | 'other'
  income_level    TEXT,                         -- 'low' | 'medium' | 'high' | 'any'
  interests       TEXT[],                       -- ['saas', 'ai', 'venture capital']
  pain_points     TEXT[],                       -- ['scaling', 'customer acquisition']
  platforms       TEXT[],                       -- ['meta', 'google', 'x', 'email']
  tone_recommendation TEXT,                     -- AI-recommended tone for this persona
  ai_generated    BOOLEAN DEFAULT false,        -- true if Gemini created this

  -- Usage tracking
  campaigns_used_count INT DEFAULT 0,

  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_personas_user_id ON public.audience_personas(user_id);


-- -------------------------------------------------------
-- 4. INTEGRATIONS
-- Tracks connected ad platform accounts.
-- -------------------------------------------------------
CREATE TABLE public.integrations (
  id              UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id         UUID REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL,

  channel         TEXT NOT NULL,                -- 'meta' | 'google' | 'x' | 'email'
  status          TEXT NOT NULL DEFAULT 'disconnected', -- 'connected' | 'disconnected' | 'error'
  account_name    TEXT,                         -- display name e.g. "Veltrix Agency"
  account_id      TEXT,                         -- platform-specific account ID
  api_key_hash    TEXT,                         -- hashed API key (never store plaintext)
  permissions     TEXT[],                       -- ['read', 'write', 'manage']
  last_synced_at  TIMESTAMPTZ,
  error_message   TEXT,                         -- if status = 'error', explain why

  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW(),

  -- One integration per channel per user
  UNIQUE(user_id, channel)
);

CREATE INDEX idx_integrations_user_id ON public.integrations(user_id);


-- -------------------------------------------------------
-- 5. CAMPAIGN ACTIVITY LOG
-- Chronological event stream for each campaign.
-- -------------------------------------------------------
CREATE TABLE public.campaign_activity (
  id              UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  campaign_id     UUID REFERENCES public.campaigns(id) ON DELETE CASCADE NOT NULL,

  event_type      TEXT NOT NULL,
    -- Possible values:
    -- 'created'          → campaign was created
    -- 'launched'         → campaign went live
    -- 'paused'           → campaign was paused
    -- 'resumed'          → campaign was resumed
    -- 'completed'        → campaign finished (end date reached or budget exhausted)
    -- 'copy_generated'   → AI generated new ad copy
    -- 'image_generated'  → AI generated ad creative image
    -- 'copy_edited'      → user manually edited the copy
    -- 'budget_changed'   → budget was modified
    -- 'audience_changed' → audience targeting was changed
    -- 'ab_test_started'  → A/B test was initiated
    -- 'ab_test_result'   → A/B test concluded with winner
    -- 'performance_alert'→ AI detected a performance anomaly

  description     TEXT NOT NULL,               -- human-readable event description
  metadata        JSONB,                       -- additional structured data

  created_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_activity_campaign_id ON public.campaign_activity(campaign_id, created_at DESC);


-- -------------------------------------------------------
-- 6. GEMINI GENERATION LOG (optional, for tracking AI usage)
-- -------------------------------------------------------
CREATE TABLE public.gemini_log (
  id              UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id         UUID REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL,

  request_type    TEXT NOT NULL,                -- 'text' | 'image' | 'persona' | 'insight'
  prompt          TEXT NOT NULL,                -- the prompt sent to Gemini
  response_text   TEXT,                         -- text response (if text generation)
  response_image_url TEXT,                      -- image URL (if image generation)
  model_used      TEXT,                         -- 'gemini-2.5-flash' or 'gemini-2.5-pro'
  tokens_used     INT,                          -- approximate token count
  latency_ms      INT,                          -- response time in milliseconds
  success         BOOLEAN DEFAULT true,

  created_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_gemini_log_user_id ON public.gemini_log(user_id, created_at DESC);

-- Enable RLS on all tables
ALTER TABLE public.campaigns ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.campaign_metrics ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.audience_personas ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.integrations ENABLE ROW LEVEL SECURITY;

ALTER TABLE public.gemini_log ENABLE ROW LEVEL SECURITY;

-- CAMPAIGNS: users can only CRUD their own
CREATE POLICY "Users manage own campaigns"
  ON public.campaigns FOR ALL
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);

-- METRICS: users can read metrics for their own campaigns
CREATE POLICY "Users read own campaign metrics"
  ON public.campaign_metrics FOR SELECT
  USING (
    campaign_id IN (
      SELECT id FROM public.campaigns WHERE user_id = auth.uid()
    )
  );

-- METRICS: system can insert (used by seed data / edge functions)
CREATE POLICY "Users insert own campaign metrics"
  ON public.campaign_metrics FOR INSERT
  WITH CHECK (
    campaign_id IN (
      SELECT id FROM public.campaigns WHERE user_id = auth.uid()
    )
  );

-- PERSONAS: users manage own
CREATE POLICY "Users manage own personas"
  ON public.audience_personas FOR ALL
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);

-- INTEGRATIONS: users manage own
CREATE POLICY "Users manage own integrations"
  ON public.integrations FOR ALL
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);

-- ACTIVITY: users read activity for their own campaigns
CREATE POLICY "Users read own campaign activity"
  ON public.campaign_activity FOR SELECT
  USING (
    campaign_id IN (
      SELECT id FROM public.campaigns WHERE user_id = auth.uid()
    )
  );

CREATE POLICY "Users insert own campaign activity"

  ON public.gemini_log FOR ALL
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);

PART 3: TRIGGERS & FUNCTIONS
============================================================

-- Auto-update the updated_at timestamp on campaigns
CREATE OR REPLACE FUNCTION public.update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER campaigns_updated_at
  BEFORE UPDATE ON public.campaigns
  FOR EACH ROW EXECUTE FUNCTION public.update_updated_at();

CREATE TRIGGER personas_updated_at
  BEFORE UPDATE ON public.audience_personas
  FOR EACH ROW EXECUTE FUNCTION public.update_updated_at();

CREATE TRIGGER integrations_updated_at
  BEFORE UPDATE ON public.integrations
  FOR EACH ROW EXECUTE FUNCTION public.update_updated_at();


-- Auto-create activity log entry when campaign status changes
CREATE OR REPLACE FUNCTION public.log_campaign_status_change()
RETURNS TRIGGER AS $$
BEGIN
  IF OLD.status IS DISTINCT FROM NEW.status THEN
    INSERT INTO public.campaign_activity (campaign_id, event_type, description, metadata)
    VALUES (
      NEW.id,
      CASE NEW.status
        WHEN 'active' THEN 'launched'
        WHEN 'paused' THEN 'paused'
        WHEN 'completed' THEN 'completed'
        ELSE 'status_changed'
      END,
      'Campaign status changed from ' || OLD.status || ' to ' || NEW.status,
      jsonb_build_object('old_status', OLD.status, 'new_status', NEW.status)
    );
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_campaign_status_change
  AFTER UPDATE ON public.campaigns
  FOR EACH ROW EXECUTE FUNCTION public.log_campaign_status_change();


-- Auto-create activity log when campaign is first created
CREATE OR REPLACE FUNCTION public.log_campaign_created()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO public.campaign_activity (campaign_id, event_type, description)
  VALUES (NEW.id, 'created', 'Campaign "' || NEW.name || '" was created');
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_campaign_created
  AFTER INSERT ON public.campaigns
  FOR EACH ROW EXECUTE FUNCTION public.log_campaign_created();
